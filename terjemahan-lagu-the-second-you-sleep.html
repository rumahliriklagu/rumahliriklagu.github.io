(+) Subject + Verb 1 + Object

(-) Subject + "don't" + Verb 1 + Object

(?) "Do" + Subject + Verb 1 + Object

(+) Subject + Verb 1 + "s/es" + Object

(?) "Does" + Subject + Verb 1 + Object

[tex] \\ [/tex]

(+) Subject + "am" + Adjective/Noun/Adverb
